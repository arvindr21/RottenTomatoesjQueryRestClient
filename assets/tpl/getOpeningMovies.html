getOpeningMovies(args, callback) 

 var rtClient = new $.rtApi({apiKey : "APIKEY"}); //Visit http://developer.rottentomatoes.com/page
 rtClient.getOpeningMovies(function (data) {     
    console.log(data); 
 });
 
 // or
 
 rtClient.getOpeningMovies({limit:9, country : 'in'},function (data) {     
    console.log(data); 
 });

Params Type Default isRequired Description
args.limit string 16 No Limits the number of movies returned
args.country string "us" No Provides localized data for the selected country (ISO 3166-1 alpha-2) if available. Otherwise, returns US data.
