getReviewsByMovieId(args, callback) 

 var rtClient = new $.rtApi({apiKey : "APIKEY"}); //Visit http://developer.rottentomatoes.com/page
 rtClient.getReviewsByMovieId({id: 770672122},function (data) {     
    console.log(data); 
 });
 
 // or
 
 rtClient.getReviewsByMovieId({id: 770672122, review_type : 'all', page_limit : 9, page : 2, country : "in"},
 function (data) {     
    console.log(data); 
 });

Params Type Default isRequired Description
args.id string -- Yes Movie ID. You can use the movies search endpoint or peruse the lists of movies/dvds to get the Movie ID
args.review_type string 'all' No 3 different review types are possible: 'all', 'top_critic' and 'dvd'. 'top_critic' shows all the Rotten Tomatoes designated top critics. 'dvd' pulls the reviews given on the DVD of the movie. 'all' as the name implies retrieves all reviews.
args.page_limit string 16 No The amount of movies in theaters to show per page
args.page string 1 No The selected page of in theaters movies
args.country string "us" No Provides localized data for the selected country (ISO 3166-1 alpha-2) if available. Otherwise, returns US data.
